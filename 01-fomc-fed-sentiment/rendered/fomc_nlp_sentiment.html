def initialize_driver():
    """Initialize Chrome driver with specific options for Mac"""
    chrome_options = Options()
    chrome_options.add_argument("--no-sandbox") # improve browser stability
    chrome_options.add_argument("--disable-dev-shm-usage") 
    chrome_options.add_argument("--start-maximized") # ensures the browser starts in full-screen
    
    # Special options for Mac
    chrome_options.add_argument('--remote-debugging-port=9222') #address potential rendering and debugging issues
    chrome_options.add_argument('--disable-gpu')
    
    driver = webdriver.Chrome(
        service=Service(ChromeDriverManager().install()),
        options=chrome_options
    )
    return driver

def scrape_fomc_videos():
    driver = initialize_driver()
    wait = WebDriverWait(driver, 20) # Wait up to 20 seconds for elements to appear
    videos = [] # List to store extracted video information
    
    try:
        # Go to Fed's channel and search for FOMC videos
        print("Navigating to Federal Reserve channel...")
        driver.get("https://www.youtube.com/@federalreserve/search?query=FOMC%20Press%20Conference")
        time.sleep(5)  # Wait for initial load
        
        # Handle cookie consent if it appears
        try:
            print("Looking for cookie consent button...")
            cookie_buttons = driver.find_elements(By.TAG_NAME, "button")
            for button in cookie_buttons:
                if "Reject all" in button.text:   #Identifies and licks the cookie rejection button
                    print("Found cookie button, clicking...")
                    button.click()
                    time.sleep(2) 
                    break
        except Exception as e:
            print(f"No cookie consent or error: {e}")
        
        # Scroll a few times to load more videos
        print("Scrolling to load more videos...")
        for i in range(5):
            driver.execute_script("window.scrollTo(0, document.documentElement.scrollHeight);")
            time.sleep(3)
            print(f"Completed scroll {i+1}")
        
        # Find all video elements
        print("Finding video elements...")
        video_elements = driver.find_elements(By.CSS_SELECTOR, "ytd-video-renderer")
        print(f"Found {len(video_elements)} total video elements")
        
        # Process each video
        for element in video_elements:
            try:
                # Get title element
                title_element = element.find_element(By.CSS_SELECTOR, "#video-title")
                title = title_element.get_attribute("title")
                
                # Only process FOMC Press Conference videos
                if "FOMC Press Conference" in title:
                    video_url = title_element.get_attribute("href")
                    
                    # Extract date from title
                    date_match = re.search(
                        r'(?:January|February|March|April|May|June|July|August|September|October|November|December)\s+\d{1,2},\s+\d{4}',
                        title
                    )
                    
                    if date_match:
                        date_str = date_match.group(0)
                        date = datetime.strptime(date_str, '%B %d, %Y') # Convert string to datetime object 
                        
                        video_info = {
                            'title': title,
                            'url': video_url,
                            'date': date
                        }
                        
                        videos.append(video_info) # Add to list of videos
                        print(f"\nFound FOMC video:")
                        print(f"Title: {title}")
                        print(f"URL: {video_url}")
                        print(f"Date: {date}")
            
            except Exception as e:
                print(f"Error processing video element: {e}")
        
        return pd.DataFrame(videos)
    
    except Exception as e:
        print(f"Major error: {e}")
        return pd.DataFrame() # Converts the list of dictionaries to a DataFrame
    
    finally:
        print("\nClosing browser...")
        driver.quit()

def main():
    print("Starting FOMC video scraper...")
    videos_df = scrape_fomc_videos()
    
    if videos_df.empty:
        print("No videos found!")
    else:
        # Save to CSV
        videos_df.to_csv('fomc_videos.csv', index=False)
        print(f"\nFound {len(videos_df)} FOMC videos. Data saved to fomc_videos.csv")
        
        # Print summary
        print("\nMost recent videos:")
        sorted_df = videos_df.sort_values('date', ascending=False)
        for _, video in sorted_df.head().iterrows():
            print(f"\nDate: {video['date']}")
            print(f"Title: {video['title']}")
            print(f"URL: {video['url']}")

if __name__ == "__main__":
    main()

def get_transcript(video_id):
    try:
        transcript = YouTubeTranscriptApi.get_transcript(video_id)
        return " ".join([entry['text'] for entry in transcript])
    except Exception as e:

def preprocess_text(text):
    # Lowercase
    text = text.lower()
    # Remove punctuation and numbers
    text = re.sub(r'[^\w\s]', '', text)
    text = re.sub(r'\d+', '', text)
    # Tokenize words
    words = text.split()
    # Remove stopwords
    words = [word for word in words if word not in stop_words]
    return words

# Preprocess the transcripts using the updated function

from transformers import AutoTokenizer, AutoModelForSequenceClassification
from transformers import pipeline

# Load FinBERT pre-trained model and tokenizer
model_name = "yiyanghkust/finbert-tone"
tokenizer = AutoTokenizer.from_pretrained(model_name)
model = AutoModelForSequenceClassification.from_pretrained(model_name)

# Initialize sentiment analysis pipeline
sentiment_analyzer = pipeline("sentiment-analysis", model=model, tokenizer=tokenizer)

# List of frequent words
frequent_words = [word for word, _ in word_counts.most_common(1000)]

# Analyze sentiment for each word
def analyze_word_sentiment(word):
    try:
        # Run FinBERT on each word (context matters more in sentences, so might adapt this for phrases later)
        result = sentiment_analyzer(word)
        return result[0]['label']  # 'Positive', 'Negative', or 'Neutral'
    except Exception as e:
        print(f"Error analyzing sentiment for {word}: {e}")
        return "Neutral"  # Default sentiment if analysis fails

# Analyze each word
sentiment_results = pd.DataFrame({
    "Word": frequent_words,
    "Sentiment": [analyze_word_sentiment(word) for word in frequent_words]
})

# Display results
print(sentiment_results)

sentiment_counts = sentiment_results['Sentiment'].value_counts()
